local function decryptLua51Bytecode(data)
    if string.byte(data, 1) == 27 and string.sub(data, 2, 4) == "Lua" then
        addLog("Lua 5.1 bytecode detected")
        local version = string.byte(data, 5)
        local format = string.byte(data, 6)
        if version == 81 then
            addLog("Lua 5.1 (0x51) format detected")
            local success, result = pcall(function()
                local f = loadstring(data)
                if f then
                    return string.dump(f), "Lua5.1"
                end
            end)
            if success and result then
                addLog("Lua 5.1 bytecode decrypted successfully")
                return result, "Lua5.1"
            end
        end
        return data, "LuaBytecode"
    end
    return data, nil
end
local function decryptLuarmor(data)
    if data:match("luarmor") or data:match("script_key") then
        addLog("Luarmor detected")
        if data:match("loadstring") then
            local url = data:match('"([^"]+)"')
            if url then
                local success, result = pcall(function()
                    return game:HttpGet(url)
                end)
                if success and result then
                    local decrypted, _ = decryptLua51Bytecode(result)
                    return decrypted, "Luarmor"
                end
            end
        end
        return data, "Luarmor"
    end
    return data, nil
end

local function decryptBase64(data)
    if data:match("^[%w%+%/%=]+$") then
        local success, result = pcall(function()
            return game:GetService("HttpService"):Base64Decode(data)
        end)
        if success and result then

    end
    return data, nil
end
local function decryptHex(data)
    if data:match("^[%x]+$") and #data % 2 == 0 then
        local hex = ""
        for i = 1, #data, 2 do
            hex = hex .. string.char(tonumber(data:sub(i, i+1), 16))
        end
        if hex:match("^[%w%p%s]+$") then
            addLog("Hex decoded")
            local decrypted, _ = decryptLua51Bytecode(hex)
            return decrypted, "Hex"
        end
    end
    return data, nil
end
local function decryptXOR(data)
    if #data > 10 then
        local xor = ""
        for i = 1, #data do
            xor = xor .. string.char(string.byte(data, i) ~ 0x55)
        end
        if xor:match("^[%w%p%s]+$") then
            addLog("XOR decoded")
            local decrypted, _ = decryptLua51Bytecode(xor)
            return decrypted, "XOR"
        end
    end
    return data, nil
end
local function decryptAll(data)
    local decoded = data
    local methods = {}
    local functions = {
        decryptLua51Bytecode,
        decryptLuarmor,
        decryptLuraph,
        decryptMoonSec,
        decryptIronBrew,
        decryptAES,
        decryptBase64,
        decryptHex,
        decryptXOR
    }
    for _, func in ipairs(functions) do
        local result, method = func(decoded)
        if method then
            table.insert(methods, method)
            decoded = result
            addLog("Decrypted with: " .. method)
        end
    end
    if #methods == 0 then
        table.insert(methods, "None")
    end
    addLog("Methods used: " .. table.concat(methods, ", "))
    return decoded, methods
end
local old_loadstring = loadstring
loadstring = function(code, chunkname)
    local decoded, methods = decryptAll(code)
    addLog("loadstring executed")
    addLog("Original code: " .. (string.sub(code, 1, 80) or "empty"))
    if #methods > 0 and methods[1] ~= "None" then
        addLog("Decrypted using: " .. table.concat(methods, ", "))
        addLog("Decrypted code: " .. (string.sub(decoded, 1, 150) or "empty"))
        notify("Decrypted successfully!")
    end
    decryptedCache[chunkname or "unknown"] = decoded
    return old_loadstring(decoded, chunkname)
end

game.HttpGet = function(self, url)
    addLog("Fetching URL: " .. tostring(url))
    local response = old_httpget(self, url)
    if response and #response > 50 then
        local decoded, methods = decryptAll(response)
        if #methods > 0 and methods[1] ~= "None" then

        end

        end
    end
    return response
end

local function scanForKeys()
    local gc = getgc(true)
    local found = {}
    for _, v in pairs(gc) do
        if type(v) == "string" then
            if string.find(v, "script_key") or string.find(v, "key") or string.find(v, "token") then
                table.insert(found, v)
            end
        end
    end
    if #found > 0 then
        addLog("Found " .. #found .. " keys")
        for i, key in ipairs(found) do
            addLog("Key: " .. key)
        end
        notify("Keys found!")
    end
    return found
end

            end
            if string.find(code, "game:HttpGet") then
                addLog("HttpGet found in: " .. name)
            end
        end
    end
end
